# FileSystemDemo
a demo for operating files(upload，edit，read)

# node-module:
npm install

# run server:
npm start

# new file page:
localhost:8080

# view file page:
localhost:8080/ViewFilePage.html

# edit file page:
localhost:8080/ViewFilePage.html?fileName={fileName}
